```components/ui/Button.tsx
import React, { forwardRef } from 'react'
import { Slot } from '@radix-ui/react-slot'
import { cn } from '../../lib/utils'
export interface ButtonProps
  extends React.ButtonHTMLAttributes<HTMLButtonElement> {
  asChild?: boolean
  variant?:
    | 'default'
    | 'destructive'
    | 'outline'
    | 'secondary'
    | 'ghost'
    | 'link'
    | 'amber'
  size?: 'default' | 'sm' | 'lg' | 'icon'
}
const Button = forwardRef<HTMLButtonElement, ButtonProps>(
  (
    {
      className,
      variant = 'default',
      size = 'default',
      asChild = false,
      ...props
    },
    ref,
  ) => {
    const Comp = asChild ? Slot : 'button'
    return (
      <Comp
        className={cn(
          'inline-flex items-center justify-center whitespace-nowrap rounded-md text-sm font-medium ring-offset-background transition-colors focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-ring focus-visible:ring-offset-2 disabled:pointer-events-none disabled:opacity-50',
          {
            'bg-slate-800 text-slate-50 hover:bg-slate-800/90':
              variant === 'default',
            'bg-red-900 text-red-50 hover:bg-red-900/90':
              variant === 'destructive',
            'border border-slate-700 bg-transparent hover:bg-slate-800 text-slate-200':
              variant === 'outline',
            'bg-slate-700 text-slate-50 hover:bg-slate-700/80':
              variant === 'secondary',
            'hover:bg-slate-800 hover:text-slate-50': variant === 'ghost',
            'text-slate-50 underline-offset-4 hover:underline':
              variant === 'link',
            'bg-amber-500 text-black hover:bg-amber-400 font-semibold':
              variant === 'amber',
            'h-10 px-4 py-2': size === 'default',
            'h-9 rounded-md px-3': size === 'sm',
            'h-11 rounded-md px-8': size === 'lg',
            'h-10 w-10': size === 'icon',
          },
          className,
        )}
        ref={ref}
        {...props}
      />
    )
  },
)
Button.displayName = 'Button'
export { Button }
```

```pages/Quotations.tsx
                    <div className="space-y-2">
                      <label className="text-sm font-medium text-slate-300">
                        Selected Items
                      </label>
                      {selectedItems.length === 0 ? (
                        <p className="text-sm text-slate-500 italic">
                          No items added yet.
                        </p>
                      ) : (
                        <div className="space-y-2">
                          {selectedItems.map((item, idx) => (
                            <div
                              key={idx}
                              className="flex items-center gap-4 p-3 rounded-lg bg-amber-500/10 border border-amber-500/20"
                            >
                              <div className="flex-1">
                                <p className="font-medium text-white">
                                  {item.name}
                                </p>
                                <p className="text-xs text-slate-400">
                                  {formatCurrency(item.rate)} x {item.days} days
                                </p>
                              </div>
                              <div className="font-mono text-white">
                                {formatCurrency(item.rate * item.days)}
                              </div>
                              <Button
                                variant="ghost"
                                size="icon"
                                className="text-red-400 hover:text-red-300"
                                onClick={() => handleRemoveItem(idx)}
                              >
                                <Trash2 className="w-4 h-4" />
                              </Button>
                            </div>
                          ))}
                        </div>
                      )}
                    </div>
                  </div>
                )}

                {step === 3 && (
                  <div className="space-y-4">
                    <div className="space-y-2">
                      <label className="text-sm font-medium text-slate-300">
                        Delivery Fee (LKR)
                      </label>
                      <Input
                        type="number"
                        value={deliveryFee}
                        onChange={(e) => setDeliveryFee(Number(e.target.value))}
                      />
                    </div>
                    <div className="space-y-2">
                      <label className="text-sm font-medium text-slate-300">
                        Discount (LKR)
                      </label>
                      <Input
                        type="number"
                        value={discount}
                        onChange={(e) => setDiscount(Number(e.target.value))}
                      />
                    </div>
                    <div className="flex items-center justify-between p-4 rounded-lg bg-white/5">
                      <span className="font-medium text-white">VAT (18%)</span>
                      <Button
                        variant={vatEnabled ? 'amber' : 'outline'}
                        size="sm"
                        onClick={() => setVatEnabled(!vatEnabled)}
                      >
                        {vatEnabled ? 'Enabled' : 'Disabled'}
                      </Button>
                    </div>

                    <div className="mt-6 space-y-2 pt-6 border-t border-white/10">
                      <div className="flex justify-between text-sm text-slate-400">
                        <span>Subtotal</span>
                        <span>{formatCurrency(subtotal)}</span>
                      </div>
                      <div className="flex justify-between text-sm text-slate-400">
                        <span>Delivery</span>
                        <span>{formatCurrency(deliveryFee)}</span>
                      </div>
                      <div className="flex justify-between text-sm text-emerald-400">
                        <span>Discount</span>
                        <span>-{formatCurrency(discount)}</span>
                      </div>
                      <div className="flex justify-between text-sm text-slate-400">
                        <span>VAT (18%)</span>
                        <span>{formatCurrency(vatAmount)}</span>
                      </div>
                      <div className="flex justify-between text-lg font-bold text-white pt-2 border-t border-white/10">
                        <span>Total</span>
                        <span>{formatCurrency(total)}</span>
                      </div>
                    </div>
                  </div>
                )}

                {step === 4 && (
                  <div className="space-y-6">
                    <div className="p-4 rounded-lg bg-emerald-500/10 border border-emerald-500/20 text-center">
                      <Check className="w-8 h-8 text-emerald-500 mx-auto mb-2" />
                      <h3 className="text-lg font-bold text-white">
                        Ready to Generate
                      </h3>
                      <p className="text-sm text-slate-400">
                        Review the details below before creating the quotation.
                      </p>
                    </div>

                    <div className="grid grid-cols-2 gap-4 text-sm">
                      <div>
                        <p className="text-slate-500">Customer</p>
                        <p className="font-medium text-white">
                          {customer || 'N/A'}
                        </p>
                      </div>
                      <div>
                        <p className="text-slate-500">Period</p>
                        <p className="font-medium text-white">
                          {startDate} to {endDate}
                        </p>
                      </div>
                    </div>

                    <div>
                      <p className="text-slate-500 text-sm mb-2">
                        Items ({selectedItems.length})
                      </p>
                      <div className="space-y-1">
                        {selectedItems.map((item, idx) => (
                          <div
                            key={idx}
                            className="flex justify-between text-sm"
                          >
                            <span className="text-slate-300">{item.name}</span>
                            <span className="text-slate-400">
                              {formatCurrency(item.rate * item.days)}
                            </span>
                          </div>
                        ))}
                      </div>
                    </div>

                    <div className="pt-4 border-t border-white/10">
                      <div className="flex justify-between text-xl font-bold text-amber-500">
                        <span>Total Amount</span>
                        <span>{formatCurrency(total)}</span>
                      </div>
                    </div>
                  </div>
                )}
              </CardContent>

              <div className="p-6 border-t border-white/5 flex justify-between bg-[#0a0f1a]/50">
                <Button
                  variant="outline"
                  onClick={() => (step > 1 ? setStep(step - 1) : resetForm())}
                >
                  {step === 1 ? 'Cancel' : 'Back'}
                </Button>

                {step < 4 ? (
                  <Button variant="amber" onClick={() => setStep(step + 1)}>
                    Next Step <ChevronRight className="w-4 h-4 ml-2" />
                  </Button>
                ) : (
                  <Button variant="amber" onClick={resetForm}>
                    Create Quotation <FileCheck className="w-4 h-4 ml-2" />
                  </Button>
                )}
              </div>
            </Card>
          </div>
        )}
      </div>
    </Layout>
  )
}
```
